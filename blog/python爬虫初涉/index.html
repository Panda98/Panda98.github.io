# 设置header
user_agent = 'Mozilla'
headers = {'User-Agent': user_agent}

url = 'www.panda.com'

# 使用requests获得网页内容

# 使用urllib2获得网页内容
# Request(url,data,headers)
request = urllib2.Request(url, headers)
response = urllib2.urlopen(request)
page = response.read()

# proxy代理设置
enable_proxy=True
proxy_handler = urllib2.ProxyHandler({'http':'http://some-proxy.com:8080'})
null_proxy_handler=urllib2.ProxyHandler({})
if enable_proxy:
    opener = urllib2.build_opener(proxy_handler)
else:
    opener = urllib2.build_opener(null_proxy_handler)
urllib2.install_opener(opener)

静态页面获取

  • 使用BeautifulSoup库来处理html页面
  soup = BeautifulSoup('<b class="boldest">Extremely bold</b>')
  # Tag
  tag = soup.b # b是指html中的<b>标签
  # 通过.name来获取tag的名字
  tag.name # b
  # 获取tag的属性
  tag['class'] # boldest
  # 获取所有属性
  tag.attrs # {u'class':u'boldest'}

遇到的Python问题

习惯了使用java，python也是很久之前学了一点，都忘了很多了…本来以为就像从java到C#一样，变通一下就是了，可是真的应用起来，发现python在很多方面的习惯和java还是挺不一样呢。虽然可能是基础，但记录一下遇到的问题，加深印象

类成员访问权限

  • public：直接写变量名或方法名

  • private：在变量名前加__

  page = [] # 这是public
  __page = [] # 这是private

类中方法中调用类成员变量或其他方法

  • 与java不同，不能直接调用，要使用self指针来调用
  def A(self):
      # ...
      
  def B(self):
      self.A

for循环

  • 与C++，java，C#不同，for循环不是由循环变量初始化、循环条件和更新三部分组成，而是类似于java中的增强for循环的结构
  pages = ['a','b','c']
  for page in pages:
      print page
  # 类似于java的普通for循环，i从0到11，注意，每一次执行完循环体，i都会更新
  # 不管是否在循环体中更新i，i都会接着上一次的i值加一
  # 所以要是想在循环体中更新，可以使用while循环
  for i in range(0,11)：
  	#...
